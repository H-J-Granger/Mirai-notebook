\index{Fermat 大定理}
\begin{dtheorem*}[Fermat\footnotemark{} 大定理，Wiles\footnotemark{}]
  对任意大于 \(2\) 的整数，都不存在正整数 \(x,y,z\) 满足 \(x^n+y^n=z^n\)。
\end{dtheorem*}

\footnotetextpre[1]{Pierre de Fermat (1607--1665)，法国数学家。}
\footnotetext{Sir Andrew John Wiles (1953--)，英国数论学家。}
